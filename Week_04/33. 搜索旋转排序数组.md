```cpp
class Solution {
public:
	int search(vector<int>& nums, int target) {
		int low = 0;
		int high = nums.size()-1;
        int res = -1;
		while (low <= high) {
			int mid = low + (high - low) / 2;
			if (nums[mid] == target) {
				res = mid;
                break;
			}
			//如果左侧有序
			if (nums[mid] >= nums[low]) {
				if (target >= nums[low] && target < nums[mid]) {
					high = mid-1;
				}
				else {
					low = mid+1;
				}

			}//如果右侧有序
			else {
				if (target > nums[mid] && target <= nums[high]) {
					low = mid + 1;
				}
				else {
					high = mid - 1;
				}

			}
		}
        return res;
	}
};
```
